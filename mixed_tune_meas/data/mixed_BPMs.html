# Remove NaN in data and change dtype to float
for k in range(0,len(data)):
    for i,clm in enumerate(data[k].T):
        for j,tmp in enumerate(clm):
            if tmp =='NaN':
                data[k][j,i] = np.nan
            else:
                data[k][j,i] = float(tmp)
